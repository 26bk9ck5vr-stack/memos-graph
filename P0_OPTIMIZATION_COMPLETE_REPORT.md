## 🔧 技术实现细节

### 查询预处理算法

```python
def preprocess_query(query: str) -> str:
    """查询预处理：拆分中文长查询为关键词"""
    import re
    
    # 策略 1: 按常见分隔符拆分
    parts = re.split(r'[\s,，.。?？!！;；:：]+', query)
    parts = [p.strip() for p in parts if p.strip()]
    
    # 策略 2: 智能拆分中英文混合
    if len(parts) == 1 and len(query) > 6:
        mixed_parts = re.findall(r'[\u4e00-\u9fa5]+|[a-zA-Z0-9]+', query)
        if len(mixed_parts) > 1:
            parts = mixed_parts
    
    # AND 逻辑连接
    return ' & '.join(f"'{p}'" for p in parts) if parts else query
```

### RRF 权重配置

```python
def rrf_fuse(hits_list, k=60, weights=None):
    if weights is None:
        weights = [4.0, 1.5, 0.5]  # FTS 权重提升到 4.0
    
    for idx, hits in enumerate(hits_list):
        weight = weights[idx] if idx < len(weights) else 1.0
        for rank, (chunk_id, score) in enumerate(hits):
            rrf = 1.0 / (k + rank + 1)
            scores[chunk_id] += rrf * score * weight
```
